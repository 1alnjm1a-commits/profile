// ── Progress Bar ──

window.addEventListener("scroll", () => {
  const pct = (scrollY / (document.body.scrollHeight - innerHeight)) * 100;
  document.getElementById("progressBar").style.width = pct + "%";
});

// ── Mobile Nav ──

function toggleNav() {
  document.getElementById("navLinks").classList.toggle("open");
}

// ── Scroll Reveal ──

const observer = new IntersectionObserver(
  (entries) => {
    entries.forEach((e) => {
      if (e.isIntersecting) {
        e.target.classList.add("visible");
        // Trigger skill bars when the why section enters view
        if (
          e.target.id === "skillBars" ||
          e.target.querySelector?.("#skillBars")
        ) {
          animateSkillBars();
        }
      }
    });
  },
  { threshold: 0.15 },
);

document.querySelectorAll(".reveal").forEach((el) => observer.observe(el));

// ── Animate Counters ──
const counterObserver = new IntersectionObserver(
  (entries) => {
    entries.forEach((e) => {
      if (e.isIntersecting) {
        document
          .querySelectorAll("[data-target]")
          .forEach((el) => animateCounter(el));
        counterObserver.disconnect();
      }
    });
  },
  { threshold: 0.5 },
);
const statsEl = document.querySelector(".hero-stats");
if (statsEl) counterObserver.observe(statsEl);

function animateCounter(el) {
  const target = +el.dataset.target;
  const dur = 1800;
  const step = 16;
  let current = 0;
  const inc = target / (dur / step);
  const timer = setInterval(() => {
    current = Math.min(current + inc, target);
    el.textContent = Math.floor(current) + (target === 100 ? "%" : "+");
    if (current >= target) clearInterval(timer);
  }, step);
}

// ── Animate Skill Bars ──
let barsAnimated = false;
function animateSkillBars() {
  if (barsAnimated) return;
  barsAnimated = true;
  document.querySelectorAll(".skill-fill").forEach((fill) => {
    fill.style.width = fill.dataset.w + "%";
  });
}

// Also trigger when skill bars become visible via scroll
const skillSection = document.getElementById("why");
if (skillSection) {
  const skillObs = new IntersectionObserver(
    (entries) => {
      if (entries[0].isIntersecting) {
        animateSkillBars();
        skillObs.disconnect();
      }
    },
    { threshold: 0.4 },
  );
  skillObs.observe(skillSection);
}

// ── Form Submit ──
function handleSubmit(btn) {
  btn.textContent = "⏳ جاري الإرسال...";
  btn.disabled = true;
  setTimeout(() => {
    btn.textContent = "✅ تم الإرسال! سأتواصل معك قريباً";
    btn.style.background = "#00cc6a";
  }, 1800);
}

// ── Smooth close nav on link click ──
document.querySelectorAll(".nav-links a").forEach((a) => {
  a.addEventListener("click", () =>
    document.getElementById("navLinks").classList.remove("open"),
  );
});
